127 private string CreateBattle(string playerDisplayName, string opponentDisplayName)
128 {
129 // Aunque playerDisplayName y opponentDisplayName no estén en la lista
130 // esperando para jugar los removemos igual para evitar preguntar si
131 // están para luego removerlos.
132 Trainer? player = WaitingList.FindTrainerByDisplayName(playerDisplayName);
133 Trainer? opponent = WaitingList.FindTrainerByDisplayName(opponentDisplayName);
134
135 if (player == null || opponent == null)
136 {
137 return $"{(player == null ? playerDisplayName : opponentDisplayName)} no está en la lista de espera";
138 }
139
140 // Remover jugadores de la lista de espera
141 WaitingList.RemoveTrainer(playerDisplayName);
142 WaitingList.RemoveTrainer(opponentDisplayName);
143
144 string actual = null;
145 int turnoRandom = new Random().Next(0, 2);
146
147 switch (turnoRandom)
148 {
149 case 0:
150 BattlesList.AddBattle(player, opponent);
151 actual = $"Empieza {player.Name}";
152 break;
153 case 1:
154 BattlesList.AddBattle(opponent, player);
155 actual = $"Empieza {opponent.Name}";
156 break;
157 }
158
159 return $"Comienza {player.Name} vs {opponent.Name}\n" +
160 $"{actual}";
161 }

169 public string StartBattle(string playerDisplayName, string? opponentDisplayName)
170 {
171 // El símbolo ? luego de Trainer indica que la variable opponent puede
172 // referenciar una instancia de Trainer o ser null.
173 Trainer? opponent;
174
175 if (!OpponentProvided() && !SomebodyIsWaiting())
176 {
177 return "No hay nadie esperando";
178 }
179
180 if (!OpponentProvided()) // && SomebodyIsWaiting
181 {
182 opponent = WaitingList.GetAnyoneWaiting();
183
184 // El símbolo ! luego de opponent indica que sabemos que esa
185 // variable no es null. Estamos seguros porque SomebodyIsWaiting
186 // retorna true si y solo si hay usuarios esperando y en tal caso
187 // GetAnyoneWaiting nunca retorna null.
188 return CreateBattle(playerDisplayName, opponentDisplayName);
189 }
190
191 // El símbolo ! luego de opponentDisplayName indica que sabemos que esa
192 // variable no es null. Estamos seguros porque OpponentProvided hubiera
193 // retorna false antes y no habríamos llegado hasta aquí.
194 opponent = WaitingList.FindTrainerByDisplayName(opponentDisplayName!);
195
196 if (!OpponentFound())
197 {
198 return $"{opponentDisplayName} no está esperando";
199 }
200
201 return CreateBattle(playerDisplayName, opponentDisplayName);
